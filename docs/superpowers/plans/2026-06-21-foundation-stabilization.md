```dart
import 'package:flutter_dotenv/flutter_dotenv.dart';

/// Centralized access to environment configuration.
///
/// Resolution order per key:
///   1. Value from the loaded `.env` file (dev mode).
///   2. Value injected via `--dart-define` (release mode).
///
/// In dev, `main.dart` calls `dotenv.load()` before any access.
/// In release, the `.env` file is absent; values come from compile-time
/// defines set with `flutter build --dart-define=KEY=value`.
class Env {
  Env._();

  /// Reads a required string value. Throws [StateError] if missing.
  static String require(String key) {
    final value = _read(key);
    if (value == null || value.isEmpty) {
      throw StateError('Missing required env var: $key');
    }
    return value;
  }

  /// Reads an optional string value. Returns null if absent.
  static String? maybe(String key) => _read(key);

  static String? _read(String key) {
    // dotenv.env is empty when no file was loaded (release builds).
    final fromFile = dotenv.maybeGet(key);
    if (fromFile != null && fromFile.isNotEmpty) return fromFile;
    // Fallback to compile-time define for release.
    return const String.fromEnvironment(key);
  }
}
```

    name: 'A',

        sb.AuthChangeEvent.initialSession,
